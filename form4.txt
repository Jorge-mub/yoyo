Private Sub btAgregar_Click()
    Dim condicion As String
    Dim tama As Long
    
    If Me.ComboBox1.Value = "" Or Me.ComboBox1.ListIndex = -1 Then
        MsgBox "Artículo no encontrado", vbExclamation, "Error"
        Exit Sub
    End If
    
    If Me.TextBox1.Value = "" Then
        MsgBox "Cantidad no encontrada", vbExclamation, "Error"
        Exit Sub
    End If
    
    If Not IsNumeric(Me.TextBox1.Value) Or InStr(Me.TextBox1.Value, ".") > 0 Or Val(Me.TextBox1.Value) <= 0 Then
        MsgBox "Cantidad no válida. Debe ingresar un número entero.", vbExclamation, "Error"
        Exit Sub
    End If
    
    If Me.cmbUsuario.Value = "" Or Me.cmbUsuario.ListIndex = -1 Then
        MsgBox "Usuario no encontrado", vbExclamation, "Error"
        Exit Sub
    End If
    
    If Val(Me.TextBox1.Value) > 100 Then
        If MsgBox("Se detectó un ingreso de unidades superior a 100, ¿Estas seguro de la cantidad ingresada?", vbYesNo + vbQuestion, "Confirmación") = vbNo Then
            Exit Sub
        End If
    End If
    
    cat = Me.ComboBox1.Value
    dog = Me.TextBox1.Value
    check = Me.CheckBox1.Value
    bird = Format(Now, "dd/mm/yyyy hh:mm:ss")
    
    tama = Me.caja1.ListCount
    
    If check = False Then
        condicion = "Nuevo"
    Else
        condicion = "Reformulado"
    End If
    
    Me.caja1.AddItem cat
    Me.caja1.List(tama, 1) = condicion
    Me.caja1.List(tama, 2) = dog
    Me.caja1.List(tama, 3) = Me.cmbUsuario.Value
    
    Me.ComboBox1.Value = ""
    Me.TextBox1.Value = ""
End Sub

Private Sub commandButton1_click()
    Dim newRow As ListRow
    Dim nFilas As Long
    Dim i As Long
    Dim tipoMovimiento As String
    Dim cantidad As Double
    
    nFilas = consumo.ListObjects("dailyConsumption").ListRows.Count + 2
    ReDim vector(1 To caja1.ListCount, 1 To 4)
    
    For i = 1 To caja1.ListCount
        consumo.ListObjects("dailyConsumption").ListRows.Add.Range(1, 1).Value = caja1.List(i - 1, 0)
        consumo.ListObjects("dailyConsumption").Range(nFilas, 2).Value = caja1.List(i - 1, 1)
        
        cantidad = Val(caja1.List(i - 1, 2))
        
        If modo = True Then
            tipoMovimiento = "Retiro"
            cantidad = -cantidad
        Else
            tipoMovimiento = "Entrada"
        End If
        
        consumo.ListObjects("dailyConsumption").Range(nFilas, 3).Value = tipoMovimiento
        consumo.ListObjects("dailyConsumption").Range(nFilas, 4).Value = cantidad
        
        With consumo.ListObjects("dailyConsumption").Range(nFilas, 5)
            .Value = Format(Now, "dd/mm/yyyy hh:mm:ss") 'CDate(caja1.List(i - 1, 0))
            .NumberFormat = "dd/mm/yyyy hh:mm:ss"
        End With
        
        consumo.ListObjects("dailyConsumption").Range(nFilas, 6).Value = caja1.List(i - 1, 3)
        
        vector(i, 1) = caja1.List(i - 1, 1)
        vector(i, 2) = caja1.List(i - 1, 0)
        vector(i, 3) = caja1.List(i - 1, 3)
        vector(i, 4) = caja1.List(i - 1, 2)
        vector(i, 5) = caja1.List(i - 1, 4)
        nFilas = nFilas + 1
    Next i
        
    Me.ComboBox1.Value = ""
    Me.TextBox1.Value = ""
    Me.CheckBox1.Value = False
    Me.cmbUsuario.Value = ""
    Call consumos
    
    Unload Me
    Almacen.Show
End Sub

Sub consumos()
    Dim tableConsumption As ListObject
    Dim tableExistencias As ListObject
    Dim cell As Range
    Dim fin As Long
    Dim i As Long

    Set tableConsumption = consumo.ListObjects("DailyConsumption")
    Set tableExistencias = Exist.ListObjects("tabExistencias")
    fin = Exist.ListObjects("tabExistencias").ListRows.Count + 1
    
    For i = 1 To UBound(vector, 1)
        For Each cell In Exist.Range("D2:D" & fin)
            If cell = vector(i, 1) Then
                If vector(i, 4) = "Nuevo" Then
                    If modo Then
                        tableExistencias.Range(cell.row, 8) = tableExistencias.Range(cell.row, 8) - vector(i, 3)
                    Else
                        tableExistencias.Range(cell.row, 8) = tableExistencias.Range(cell.row, 8) + vector(i, 3)
                    End If
                Else
                    If modo Then
                        tableExistencias.Range(cell.row, 9) = tableExistencias.Range(cell.row, 9) - vector(i, 3)
                    Else
                        tableExistencias.Range(cell.row, 9) = tableExistencias.Range(cell.row, 9) + vector(i, 3)
                    End If
                End If
            End If
        Next cell
    Next i
    
    Me.caja1.Clear
End Sub

Sub FiltrarCaja2(ByVal sector As String)
    Dim tbl As ListObject
    Dim vData As Variant
    Dim i As Long, k As Long
    Dim tipoSel As String, grupoSel As String
    Dim tipoFila As String, grupoFila As String, artFila As String, artFila2 As String
    Dim arrArt() As Variant
    Dim cuenta As Long
    
    Set tbl = Exist.ListObjects("tabExistencias")
    If tbl.ListRows.Count < 1 Then Exit Sub
    vData = tbl.DataBodyRange.Value
    
    tipoSel = Me.cmbFiltroTipo.Value
    grupoSel = Me.cmbFiltroGrupo.Value
    
    cuenta = 0
    For i = 1 To UBound(vData, 1)
        tipoFila = CStr(vData(i, 1))
        grupoFila = CStr(vData(i, 6))
        If (tipoSel = "" Or tipoFila = tipoSel) And (grupoSel = "" Or grupoFila = grupoSel) Then
            cuenta = cuenta + 1
        End If
    Next i
    
    If cuenta = 0 Then
        bloqueaEventos = True
        If sector = "Tipo" Then
            Me.cmbFiltroGrupo.Value = ""
            grupoSel = ""
        ElseIf sector = "Grupo" Then
            Me.cmbFiltroTipo.Value = ""
            tipoSel = ""
        End If
        bloqueaEventos = False
        
        cuenta = 0
        For i = 1 To UBound(vData, 1)
            tipoFila = CStr(vData(i, 1))
            grupoFila = CStr(vData(i, 6))
            If (tipoSel = "" Or tipoFila = tipoSel) And (grupoSel = "" Or grupoFila = grupoSel) Then
                cuenta = cuenta + 1
            End If
        Next i
    End If
    
    Me.caja2.Clear
    If cuenta > 0 Then
        ReDim arrArt(1 To cuenta, 1 To 2)
        k = 1
        For i = 1 To UBound(vData, 1)
            tipoFila = CStr(vData(i, 1))
            grupoFila = CStr(vData(i, 6))
            artFila = CStr(vData(i, 4))
            artFila2 = CStr(vData(i, 20))
            If (tipoSel = "" Or tipoFila = tipoSel) And (grupoSel = "" Or grupoFila = grupoSel) Then
                arrArt(k, 1) = artFila
                arrArt(k, 2) = artFila2
                k = k + 1
            End If
        Next i
        Me.caja2.List = arrArt
    End If
End Sub
